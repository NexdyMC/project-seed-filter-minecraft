#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
    // Seed the random number generator using the current time
    // This is typically called only once per program run
    srand(time(NULL));

    // Define the minimum and maximum values for an 8-digit number
    long min = 10000000;
    long max = 99999999;

    // Generate a random number within the specified range
    // The formula for a range [min, max] is: (rand() % (max - min + 1)) + min
    long random_number = (rand() % (max - min + 1)) + min;

    // Print the generated 8-digit random number
    printf("Generated 8-digit random number: %ld\n", random_number);

    return 0;
}
